Display Name of the salesperson, customer name and amount of sales. 
Ans select customers.cname, salespeople.sname, sum(amt)  from customers join salespeople on customers.SNUM = salespeople.SNUM inner join orders on customers.cnum = orders.cnum group by customers.cname, salespeople.sname;

List employee name and name of managers.
Ans SELECT  e.ename AS 'Employee', m.ename AS 'Manager' FROM emp e LEFT JOIN emp m  ON e.mgr = m.empno;

Display details of all departments and the corresponding employees 
Ans SELECT d.deptno, d.dname, d.loc, e.empno, e.ename,  e.job, e.sal FROM dept d
LEFT JOIN emp e ON d.deptno = e.deptno ORDER BY d.deptno;

Display all customer names with their order details (order number and ship date) for the orders shipped in 1987.
Ans SELECT c.name, o.ordid, o.shipdate FROM customer c JOIN orders o ON c.custid = o.custid WHERE YEAR(o.shipdate) = 1987;

Display salesman names and names of customers they are dealing with.
Ans SELECT s.sname AS 'Salesman',  c.cname AS 'Customer' FROM salespeople s
JOIN customers  c ON s.snum = c.snum;

Display salesman who did not earn commission, with customer names whom they are dealing with.	
Ans SELECT  s.sname AS 'Salesman',  c.cname AS 'Customer' FROM salespeople s
JOIN customers c ON s.snum = c.snum WHERE s.comm IS NULL OR s.comm = 0;

Display salesman names, salary, commission and their salary grades.
Ans SELECT s.sname AS 'Salesman', s.sal AS 'Salary', s.comm AS 'Commission' g.grade AS 'Salary Grade' FROM salesman s JOIN salgrade g ON s.sal BETWEEN g.losal AND g.hisal;

List out the common jobs in Research and Accounting Departments in ascending order.
Ans SELECT DISTINCT e.job FROM emp e JOIN dept d ON e.deptno = d.deptno WHERE d.dname = 'RESEARCH' INTERSECT SELECT DISTINCT e.job FROM emp e JOIN dept d  ON e.deptno = d.deptno WHERE d.dname = 'ACCOUNTING' ORDER BY job;
